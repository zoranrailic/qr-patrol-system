void showAlertDialog(
  String? message,
  String title,
  AlertType alertType,
  GlobalKey<ScaffoldState> scaffoldKey, {
  required bool isAnyButton,
}) {
  if (isAnyButton == true) {
    Alert(
      context: scaffoldKey.currentContext!,
      style: alertStyle,
      type: alertType,
      title: title,
      desc: message,
      buttons: [
        DialogButton(
          onPressed: () => Navigator.pop(scaffoldKey.currentContext!),
          width: 120,
          child: const Text(
            okText,
            style: TextStyle(color: Colors.white, fontSize: 20),
          ),
        )
      ],
    ).show();
  } else {
    Alert(
      context: scaffoldKey.currentContext!,
      style: alertStyle,
      type: alertType,
      title: title,
      desc: message,
      buttons: [],
    ).show();
  }
}
